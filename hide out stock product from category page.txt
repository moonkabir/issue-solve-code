add_filter('woocommerce_product_query_meta_query', 'merchshark_filter_product_query_meta_query', 10, 2);
function merchshark_filter_product_query_meta_query($meta_query, $query)
{
	if (!is_shop()) {
		$meta_query[] = array(
			'key' => '_stock_status',
			'value' => 'outofstock',
			'compare' => '!=',
		);
	}
	return $meta_query;
}
